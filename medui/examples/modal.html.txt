<!-- ✅ JS ควบคุม Modal -->
<script>
    const modal = document.getElementById("demoModal");
    document.getElementById("openModal").onclick = () => modal.classList.add("show");
    document.getElementById("closeModal").onclick = () => modal.classList.remove("show");
    document.getElementById("closeModal2").onclick = () => modal.classList.remove("show");
    modal.onclick = e => { if(e.target === modal) modal.classList.remove("show") }; // ปิดเมื่อคลิกฉากหลัง
</script>
